package cn.zvc.two;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

public class CollectionClassifier {

    public static void main(String[] args) {
        Collection<?>[] collections = { new HashSet<String>(), new ArrayList<String>(),
                new HashMap<String, String>().values() };
        for(Collection<?> c : collections){
            System.out.println(classify(c));
        }
    }

    public static String classify(Set<?> s) {
        return "set";
    }

    public static String classify(List<?> list) {
        return "list";
    }

    public static String classify(Collection<?> c) {
        return "unknow collection";
    }
}

Set和List的重载

package cn.zvc.two;

import java.util.ArrayList;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

public class SetList {

    public static void main(String[] args) {
        Set<Integer> set = new HashSet<Integer>();
        List<Integer> list = new ArrayList<Integer>();
        for(int i=-3;i<3; i++){
            set.add(i);
            list.add(i);
        }

        for(int i=0;i<3; i++){
            set.remove(i);//boolean java.util.Set.remove(Object o)  自动装箱  删除元素0,1,2
            list.remove(i); //Integer java.util.List.remove(int index)  删除索引
            //boolean java.util.List.remove(Object o)
        }

        System.out.println(set + " " + list);//[-1, -2, -3] [-2, 0, 2]

    }
}

Nginx集群 负载均衡策略以及配置

By zvc

发表于 2018-04-26

Nginx集群 负载均衡策略以及配置

轮询（Round Robin）根据Nginx配置文件中的顺序，依次把客户端的Web请求分发到不同的后端服务器

最少连接 请求被转发到连接数最少的服务器上

只是在upstream节点添加least_conn配置

IP地址哈希

只是在upstream节点添加ip_hash配置

权重

基于权重的负载均衡即Weighted Load Balancing

上面例子在服务器地址和端口后加weight=2配置，意味着每4个请求有2个分发到第一个服务器，第3个分发到第二个服务器，第3个请求分发到第三个服务器

ps:基于权重的负载均衡和基于IP地址哈希的负载均衡可以组合在一起使用。

synchronized和volatile比较

By zvc

发表于 2018-04-26

synchronized和volatile比较

  • volatile不需要枷锁，比synchronized更加轻量级，不会阻塞线程；
  • 从内存可见性角度讲，volatile读相当于加锁，volatile写相当于解锁
  • synchronized既能保证可见性，又能保证原子性，而volatile只能保证可见性，无法保证原子性。

对64位（long、double）变量的读写可能不是院子操作：

  • Java内存模型允许JVM将没有被volatile修饰的64位数据类型的读写操作划分为两次32位的读写操作来进行

导致问题：有可能会出现读取到“半个变量”的情况

解决方法：加volatile关键字
